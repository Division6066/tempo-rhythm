7. SEARCH IMPROVEMENTS (convex/search.ts)
```typescript
   export const advancedSearch = query({
     args: {
       query: v.string(),
       type: v.optional(v.union(v.literal("tasks"), v.literal("notes"))),
       status: v.optional(v.string()),
       priority: v.optional(v.string()),
       projectId: v.optional(v.id("projects")),
     },
     handler: async (ctx, args) => {
       const userId = await getAuthUserId(ctx);
       if (!userId) return [];
       
       const lowerQuery = args.query.toLowerCase();
       
       const results: any[] = [];
       
       if (args.type === "tasks" || !args.type) {
         const tasks = await ctx.db
           .query("tasks")
           .withIndex("by_user", (q) => q.eq("userId", userId))
           .collect();
         
         const filtered = tasks.filter((task) => {
           const matchesQuery =
             task.title.toLowerCase().includes(lowerQuery) ||
             (task.description?.toLowerCase().includes(lowerQuery) ?? false);
           
           const matchesStatus = !args.status || task.status === args.status;
           const matchesPriority = !args.priority || task.priority === args.priority;
           const matchesProject = !args.projectId || task.projectId === args.projectId;
           
           return matchesQuery && matchesStatus && matchesPriority && matchesProject;
         });
         
         results.push(...filtered.map((t) => ({ type: "task", ...t })));
       }
       
       if (args.type === "notes" || !args.type) {
         const notes = await ctx.db
           .query("notes")
           .withIndex("by_user", (q) => q.eq("userId", userId))
           .collect();
         
         const filtered = notes.filter((note) => {
           const matchesQuery =
             note.title.toLowerCase().includes(lowerQuery) ||
             (note.content?.toLowerCase().includes(lowerQuery) ?? false);
           
           const matchesProject = !args.projectId || note.projectId === args.projectId;
           
           return matchesQuery && matchesProject;
         });
         
         results.push(...filtered.map((n) => ({ type: "note", ...n })));
       }
       
       return results;
     },
   });
```
